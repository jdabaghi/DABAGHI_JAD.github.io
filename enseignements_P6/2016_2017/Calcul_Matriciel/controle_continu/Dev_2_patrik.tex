\noindent \textbf{Questions du cours :} \\
\begin{enumerate}[font=\bfseries,label=(Q\arabic*)]
\item  Donner la définition d'\textit{une application linéaire} $F: \R^2 \to \R^2$.
\item  Donner la définition d'une application linéaire $P: \R^3 \to \R^3$ qui est \textit{une projection} dans $\mathbb{R}^3$.

\end{enumerate}

\textbf{Exercice 4:}
\end{center} 
On considère l'application suivante:
 $f : \R^3 \longrightarrow \R^3$, $\begin{pmatrix} x \\ y\\z  \end{pmatrix} \longmapsto \begin{pmatrix} y+z \\ x+y+z\\x  \end{pmatrix}$
 
 \begin{enumerate}
 \item Montrez que $f$ est linéaire.
 \item Donner l'expression de $F$, la matrice de $f$ dans la base canonique de $\R^3$.
 \item Déterminer l'ensemble ker($f$)$= \lbrace x \in \R^3\,|\, f(x) = 0 \rbrace$
 \item Déterminer l'ensemble im($f$)$= \lbrace f(x)\,,\,x \in \R^3\ \rbrace$
 \end{enumerate}
